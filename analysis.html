import pandas as pd import matplotlib.pyplot as plt # Load data df = pd.read_csv('employee_performance.csv') # Calculate R&D count rd_count = df[df['department'] == 'R&D'].shape[0] print(f"R&D Dept Count: {rd_count}") # Plot plt.figure(figsize=(10, 6)) df['department'].value_counts().plot(kind='bar') plt.title('Department Distribution') plt.xlabel('Department') plt.ylabel('Frequency') plt.show()
